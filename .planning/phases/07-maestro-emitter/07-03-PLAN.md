The step-mapper signature:

```typescript
import type { Spec } from "../../model/index.ts";
import type { TestFlowStep } from "../../model/spec.ts";
import type { Diagnostic } from "../../primitives/diagnostic.ts";
import type { JsonPointer } from "../../primitives/path.ts";
import { pathToJsonPointer } from "../../primitives/path.ts";

export type StepMapResult =
  | { ok: true; tapOnId: string; customActionName?: string }
  | { ok: false; diagnostic: Diagnostic };

export function mapStep(
  step: TestFlowStep,
  spec: Spec,
  flowIndex: number,
  stepIndex: number,
): StepMapResult {
  // 1. Find the screen
  const screen = spec.screens.find((s) => s.id === step.screen);
  if (!screen) {
    // Should have been caught by crossReferencePass, but defend anyway
    return {
      ok: false,
      diagnostic: {
        code: "MAESTRO_UNRESOLVED_SCREEN",
        severity: "error",
        path: pathToJsonPointer(["test_flows", flowIndex, "steps", stepIndex, "screen"]),
        message: `screen "${step.screen}" not found`,
      },
    };
  }

  // 2. Resolve testID — walk content variant only (D-106)
  const testID = findTestIDForAction(screen, step.action);
  if (testID === null) {
    return {
      ok: false,
      diagnostic: {
        code: "MAESTRO_MISSING_TESTID",
        severity: "error",
        path: pathToJsonPointer(["test_flows", flowIndex, "steps", stepIndex, "action"]),
        message: `no testID found for action "${step.action}" on screen "${step.screen}"`,
      },
    };
  }

  // 3. Determine custom action name (for comment emission)
  const action = spec.actions[step.action];
  const customActionName = action?.kind === "custom" ? action.name : undefined;

  return { ok: true, tapOnId: testID, customActionName };
}
```

EmitResult type and emitMaestroFlows function:

```typescript
export type EmitResult =
  | { ok: true; flows: Array<{ name: string; ios: string; android: string }> }
  | { ok: false; diagnostics: Diagnostic[] };

export function emitMaestroFlows(spec: Spec): EmitResult {
  if (!spec.test_flows || spec.test_flows.length === 0) {
    return { ok: true, flows: [] };
  }
  const flows: Array<{ name: string; ios: string; android: string }> = [];
  const diagnostics: Diagnostic[] = [];
  for (let fi = 0; fi < spec.test_flows.length; fi++) {
    const flow = spec.test_flows[fi]!;
    const result = emitMaestroFlow(flow, spec, fi);
    if (!result.ok) {
      diagnostics.push(...result.diagnostics);
    } else {
      flows.push({ name: flow.name, ios: result.ios, android: result.android });
    }
  }
  if (diagnostics.length > 0) return { ok: false, diagnostics };
  return { ok: true, flows };
}
```

Private `emitMaestroFlow(flow, spec, fi)` function:
1. Filter steps for each platform via `filterStepsByPlatform`
2. For each step in the filtered list, call `mapStep(step, spec, fi, si)`
3. Collect diagnostics; if any → `{ ok: false, diagnostics }`
4. Build the steps array for each platform (plain JS objects):
   - Prepend `"launchApp"` as the first element (bare string in YAML list)
   - For each step result: push `{ tapOn: { id: tapOnId } }` as an object
   - For custom action steps: the comment `# custom action: <name>` cannot be embedded in YAML.stringify objects. Use a string-joining approach:
     - Build a `rawLines: string[]` for each platform
     - Start with `rawLines = ["- launchApp"]`
     - For each step: push `YAML.stringify({ tapOn: { id } }).trimEnd()` prefixed with `- ` (YAML list item)
     - For custom: prepend a `# custom action: <name>` line before the tapOn line
     - Join all lines with `\n`
5. Assemble final YAML:
   ```
   # Replace com.example.app with your bundle identifier\n
   appId: com.example.app\n
   ---\n
   <steps lines>
   ```
   Use `YAML.stringify({ appId: "com.example.app" }).trimEnd()` for the header object line.

Note on YAML structure for `tapOn: { id }`:
```typescript
// YAML.stringify({ tapOn: { id: "my_btn" } }) produces:
// tapOn:
//   id: my_btn
// So prefix with "- " to make it a list item:
// - tapOn:
//     id: my_btn
// But YAML.stringify returns multi-line with indentation that needs "- " prefix on first line only.
// Pattern: split lines, prefix first with "- ", indent remaining with "  "
function yamlListItem(obj: object): string {
  const lines = YAML.stringify(obj).trimEnd().split("\n");
  return ["- " + lines[0]!, ...lines.slice(1).map((l) => "  " + l)].join("\n");
}
```
