    button.ts:
    ```typescript
    // renderButton per D-34:
    //   variant=primary   → [[ label ]]   (6-char overhead)
    //   variant=secondary → [ label ]     (4-char overhead; DEFAULT when omitted)
    //   variant=text      → label          (0-char overhead)
    // D-42: action + testID NEVER rendered in persisted output.
    import type { ComponentNode } from "../../../model/component.ts";
    import { padRight } from "../layout.ts";
    import { truncate } from "../overflow.ts";

    export function renderButton(
      node: Extract<ComponentNode, { kind: "Button" }>,
      width: number,
    ): string[] {
      const variant = node.variant ?? "secondary";
      let glyph: string;
      if (variant === "primary") {
        const label = truncate(node.label, Math.max(1, width - 6));
        glyph = `[[ ${label} ]]`;
      } else if (variant === "text") {
        glyph = truncate(node.label, width);
      } else {
        // secondary (default)
        const label = truncate(node.label, Math.max(1, width - 4));
        glyph = `[ ${label} ]`;
      }
      return [padRight(truncate(glyph, width), width)];
    }
    ```

    import type { ComponentNode } from "../../../model/component.ts";

      return [padRight(truncate(line, width), width)];
    }

    text-field.ts:
    ```typescript
    // renderTextField per D-34: `label: ________________` — colon-separator,
    // underscore run fills to width. Phase-3 v1 ignores bindsTo/placeholder;
    // field renders as empty (no live data to resolve at render time).
    // D-42: action + testID hidden.
    import type { ComponentNode } from "../../../model/component.ts";
    import { padRight } from "../layout.ts";
    import { truncate } from "../overflow.ts";

    export function renderTextField(
      node: Extract<ComponentNode, { kind: "TextField" }>,
      width: number,
    ): string[] {
      const separator = ": ";
      const minUnderscores = 4; // visible-field floor
      const labelBudget = Math.max(1, width - separator.length - minUnderscores);
      const label = truncate(node.label, labelBudget);
      const prefix = `${label}${separator}`;
      const underscoreCount = Math.max(minUnderscores, width - prefix.length);
      const line = prefix + "_".repeat(underscoreCount);
      return [padRight(truncate(line, width), width)];
    }
    ```

    // renderSegmentedControl per D-34: `< opt1 | opt2 | opt3 >`. Phase-3 v1
