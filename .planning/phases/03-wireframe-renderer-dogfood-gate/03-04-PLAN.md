    ```typescript
    // renderSpacer per D-36: size → blank-line count (sm=1, md=2 default, lg=3).
    import type { ComponentNode } from "../../../model/component.ts";

    export function renderSpacer(
      node: Extract<ComponentNode, { kind: "Spacer" }>,
      width: number,
    ): string[] {
      const count = node.size === "sm" ? 1 : node.size === "lg" ? 3 : 2;
      const blank = " ".repeat(Math.max(0, width));
      const out: string[] = [];
      for (let i = 0; i < count; i++) out.push(blank);
      return out;
    }
    ```

    src/emit/wireframe/components/image.ts:
    ```typescript
    // renderImage per D-Claude: default 3-line `+--IMG--+` box at 10 cols wide.
    // Width < 8 cols → fallback to inline `[img:alt]`.
    // Missing alt (defensive; Phase-1 schema requires alt) → `(no alt)` inline.
    import type { ComponentNode } from "../../../model/component.ts";
    import { padRight } from "../layout.ts";
    import { truncate } from "../overflow.ts";

    const DEFAULT_IMG_WIDTH = 10;

    export function renderImage(
      node: Extract<ComponentNode, { kind: "Image" }>,
      width: number,
    ): string[] {
      const alt = node.alt && node.alt.length > 0 ? node.alt : "(no alt)";

      // Inline fallback for narrow widths.
      if (width < 8) {
        const glyph = `[img:${alt}]`;
        return [padRight(truncate(glyph, width), width)];
      }

      const boxWidth = Math.min(DEFAULT_IMG_WIDTH, width);
      // Top: +--IMG---+, bottom: +--------+, padded to box width internally.
      const topInner = "IMG";
      const topLen = boxWidth - 2; // minus two `+` corners
      const top = `+${"-".repeat(Math.max(0, Math.floor((topLen - topInner.length) / 2)))}${topInner}${"-".repeat(Math.max(0, topLen - topInner.length - Math.floor((topLen - topInner.length) / 2)))}+`;
      const bottom = `+${"-".repeat(Math.max(0, boxWidth - 2))}+`;
      const altInner = boxWidth - 4; // minus `| ` + ` |`
      const altLine = `| ${padRight(truncate(alt, altInner), altInner)} |`;

      // If box is narrower than full width, pad each line rightward to width.
      return [padRight(top, width), padRight(altLine, width), padRight(bottom, width)];
    }
    ```
