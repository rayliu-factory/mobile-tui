parseSpecFile real contract (replaces Wave-0 stub body in src/serialize/parse.ts):
```typescript
import type { Document } from "yaml";
import type { Spec } from "../model/index.ts";
import type { Diagnostic } from "./diagnostics.ts";
import type { AstHandle } from "./ast-handle.ts";

export interface ParseResult {
  spec: Spec | null;
  astHandle: AstHandle | null;
  diagnostics: Diagnostic[];
  body: string;  // body bytes verbatim from file.orig
}

/**
 * Read + parse a .spec.md file into a validated Spec + opaque AstHandle.
 *
 * NEVER THROWS on schema errors — returns Diagnostic[] for:
 *   - Missing frontmatter delimiter (!parsed.hasFrontmatter) → SERDE_MISSING_DELIMITER error
 *   - .strict() violations (including __proto__ etc.) → Phase-1 .strict() error
 *   - Cross-ref failures (unresolved actions, testID collisions) → Stage-B errors
 *
 * DOES THROW on:
 *   - ENOENT / EACCES from fs.readFile (unrecoverable IO)
 *   - YAML.parseDocument throws on syntactically invalid YAML
 *   - Path ends in `.tmp` (authoring mistake; RESEARCH Open Q#4)
 */
export async function parseSpecFile(path: string): Promise<ParseResult>;
```

Fixture byte-identical round-trip assertion shape:
```typescript
it.each(FIXTURES)("round-trips %s byte-identically", async (fixturePath) => {
  const abs = resolve(fixturePath);
  const originalBytes = await fs.readFile(abs);
  const { spec, astHandle, diagnostics } = await parseSpecFile(abs);
  const errors = diagnostics.filter((d) => d.severity === "error");
  expect(errors).toEqual([]);
  expect(spec).not.toBeNull();
  expect(astHandle).not.toBeNull();
  const tmpTarget = resolve("tests/tmp", `${randomUUID()}.spec.md`);
  const { written } = await writeSpecFile(tmpTarget, spec!, astHandle!);
  expect(written).toBe(true);
  const rtBytes = await fs.readFile(tmpTarget);
  expect(Buffer.equals(originalBytes, rtBytes)).toBe(true);
});
```

      diagnostics.push(...yaml11GotchaLint(parsed.doc));

      const { spec, diagnostics: validationDiagnostics } = validateSpec(partition.knownSubset);
      diagnostics.push(...validationDiagnostics);

      const astHandle: AstHandle = {

        bodyBytes: parsed.bodyBytes,
        origBytes: parsed.origBytes,

        frontmatterStart: parsed.frontmatterStart,
        frontmatterEnd: parsed.frontmatterEnd,
        closingDelimiterTerminator: parsed.closingDelimiterTerminator,
        hasFrontmatter: parsed.hasFrontmatter,
      };

      return {

        astHandle: spec ? astHandle : null,
        diagnostics,
        body: parsed.bodyBytes,

                `unquoted YAML-1.1 boolean-like scalar "${scalar.value}" — quote it or use an explicit boolean`,
              ),
            );
          }
        },
      });
      return out;
