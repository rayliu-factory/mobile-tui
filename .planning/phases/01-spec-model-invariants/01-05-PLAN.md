    ```ts
    // Screen schema (SPEC-02, SPEC-05, SPEC-10).
    //
    // Structure:
    //   id: ScreenId (snake_case)
    //   title: non-empty string
    //   kind: "regular" | "overlay"  (resolves RESEARCH Open Q#3 — direct cross-ref for present.overlay)
    //   back_behavior: optional here; required for non-root screens (cross-ref Plan 06)
    //   variants: ScreenVariants wired to real ComponentNodeSchema
    //   acceptance: optional string[] of prose one-liners (SPEC-10)
    import { z } from "zod";
    import { ScreenIdSchema } from "../primitives/ids.ts";
    import { BackBehaviorSchema } from "./back-behavior.ts";
    import { createScreenVariantsSchema } from "./variant.ts";
    import { ComponentNodeSchema } from "./component.ts";

    // Wire the variant factory with the real ComponentNodeSchema — this is where
    // Plans 03 and 04 converge.

    export const SCREEN_KINDS = ["regular", "overlay"] as const;
    export type ScreenKind = (typeof SCREEN_KINDS)[number];

    export const ScreenSchema = z.object({
      id: ScreenIdSchema,
      title: z.string().min(1, "screen title must be non-empty"),
      kind: z.enum(SCREEN_KINDS),
      back_behavior: BackBehaviorSchema.optional(),
      variants: ScreenVariantsWithComponentsSchema,
      acceptance: z.array(z.string().min(1, "acceptance line must be non-empty")).optional(),
    }).strict();

    export type Screen = z.infer<typeof ScreenSchema>;
    ```

    ```ts
    // src/model/navigation.ts
    // Navigation graph — SPEC-03.
    //
    // root: the starting screen. Cross-ref (Plan 06) ensures it exists in screens[].
    // edges: directed links with an ActionId trigger and a visual transition.
    //
    // back_behavior lives on Screen, not NavEdge, because it describes how the
    // user leaves a screen regardless of which edge brought them there.
    import { z } from "zod";
    import { ScreenIdSchema, ActionIdSchema } from "../primitives/ids.ts";

    export const TRANSITIONS = ["push", "modal", "sheet", "replace", "none"] as const;
    export type NavTransition = (typeof TRANSITIONS)[number];

    export const NavEdgeSchema = z.object({
      from: ScreenIdSchema,
      to: ScreenIdSchema,
      trigger: ActionIdSchema,
      transition: z.enum(TRANSITIONS).optional(),
    }).strict();
    export type NavEdge = z.infer<typeof NavEdgeSchema>;

    export const NavigationGraphSchema = z.object({
      root: ScreenIdSchema,
      edges: z.array(NavEdgeSchema),
    }).strict();
    export type NavigationGraph = z.infer<typeof NavigationGraphSchema>;
    ```

    ```ts
    // src/model/spec.ts
    // Top-level Spec schema — the root composition.
    //
    // Phase 1 is STRICT: unknown top-level keys are rejected. Phase 2 will add
    // an _unknown: bucket for forward-compat (SPEC-08), but that's the serializer's
    // concern, not the model's. Here, the schema is the closed contract.
    //
    // schema version: pinned to SCHEMA_VERSION ("mobile-tui/1"). Specs claiming any
    // other version land in the migration runner (Plan 07) first.
    import { z } from "zod";
    import { SCHEMA_VERSION } from "./version.ts";
    import { ScreenSchema } from "./screen.ts";
    import { NavigationGraphSchema } from "./navigation.ts";
    import { ActionsRegistrySchema } from "./action.ts";
    import { DataModelSchema } from "./data.ts";

    export const SpecSchema = z.object({
      // Accept ONLY the current schema version string. Migration runner (Plan 07)
      // upgrades older specs BEFORE SpecSchema.safeParse is called.
      schema: z.literal(SCHEMA_VERSION),

      // Screens — at least one; cross-ref (Plan 06) validates uniqueness + root-existence.
      screens: z.array(ScreenSchema).min(1, "spec must declare at least one screen"),

      // Actions registry (D-10, D-11). Cross-ref validates navigate.screen / submit.entity / etc.
      actions: ActionsRegistrySchema,

      // Data model (SPEC-04).
      data: DataModelSchema,

      // Navigation graph (SPEC-03). Root screen existence enforced by cross-ref.
      navigation: NavigationGraphSchema,
    }).strict();  // Phase 2 relaxes via _unknown: bucket; Phase 1 is strict.

    export type Spec = z.infer<typeof SpecSchema>;
    ```
