// D-83: 20% / 40% / 40% split; min 15/30/30; collapse preview if total < 75
function calcPaneWidths(total: number): [number, number, number] {
  if (total < 75) {
    const screens = Math.max(15, Math.floor(total * 0.2));
    return [screens, total - screens, 0]; // preview=0 means collapsed
  }
  const screens = Math.max(15, Math.floor(total * 0.2));
  const preview = Math.max(30, Math.floor(total * 0.4));
  const inspector = total - screens - preview; // absorbs rounding remainder
  return [screens, inspector, preview];
}
```

    // src/canvas/focus-fsm.ts
    // Focus state machine for the canvas 3-pane layout (D-77).
    // Pure functions: no side effects, no imports from pi-tui.
    export type FocusState = "screens" | "inspector" | "preview" | "palette";

    export const FOCUS_CYCLE: FocusState[] = ["screens", "inspector", "preview"];

    /**
     * Advance or reverse the focus cycle (D-77).
     * "palette" state always resolves back to "screens" (palette close → restore base).
     */
    export function nextFocus(state: FocusState, reverse = false): FocusState {
      if (state === "palette") return "screens";
      const idx = FOCUS_CYCLE.indexOf(state);
      if (idx === -1) return "screens";
      const next = reverse
        ? (idx - 1 + FOCUS_CYCLE.length) % FOCUS_CYCLE.length
        : (idx + 1) % FOCUS_CYCLE.length;
      return FOCUS_CYCLE[next]!;
    }
    ```

    **HorizontalLayout.render — row compositing:**
    ```typescript
    render(width: number): string[] {
      const activePanes = this.panes.filter(p => p.width > 0);
      const lineArrays = activePanes.map(p => p.component.render(p.width));
      const maxLines = Math.max(0, ...lineArrays.map(a => a.length));
      const result: string[] = [];
      for (let row = 0; row < maxLines; row++) {
        let line = "";
        for (let col = 0; col < activePanes.length; col++) {
          const paneLines = lineArrays[col]!;
          const paneLine = paneLines[row] ?? " ".repeat(activePanes[col]!.width);
          // truncateToWidth pad=true: fills to exact width, ANSI-safe
          line += truncateToWidth(paneLine, activePanes[col]!.width, "", true);
        }
        result.push(line);
      }
      return result;
    }
    ```
