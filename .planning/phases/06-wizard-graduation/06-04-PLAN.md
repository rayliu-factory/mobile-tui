<objective>
Implement the wizard's interactive form pane: FormPane orchestrates all 8 step forms,
handles step navigation (Tab→advance, Esc→retreat), pre-populates inputs from spec on
re-entry, and shows the step indicator at the top.

Steps 3 (Screens) and 5 (Data) have stateful add-one-by-one list UX (D-92) with their
own sub-components: ScreensStep and DataStep. All other steps use a single Input field.

The critical Pitfall 5 from RESEARCH.md is handled here: ScreensStep returns
`{ kind: "consumed" }` on Enter (add screen) vs `{ kind: "advance", args }` on Tab
(finish step) — so FormPane only advances when it receives an advance signal.

Purpose: FormPane is the most complex wizard component — it must orchestrate step routing,
input accumulation, and StepAction dispatching correctly for all 8 steps.

Output: FormPane + ScreensStep + DataStep — fully functional step form rendering and input.
</objective>

```typescript
handleInput(data: string): StepAction {
  if (data === "\r" || data === "\n") {  // Enter — add item
    const trimmed = this.inputValue.trim();
    if (trimmed.length > 0) { this.items.push(trimmed); this.inputValue = ""; }
    return { kind: "consumed" };
  }
  if (data === "\t") {  // Tab — finish step
    if (this.items.length === 0 && this.requiresMin1) {
      this.error = "At least one screen is required.";
      return { kind: "consumed" };
    }
    return { kind: "advance", args: this.items };
  }
  if (data === "\x7f" || data === "\b") {  // Backspace
    if (this.inputValue.length === 0 && this.items.length > 0) { this.items.pop(); }
    else { this.inputValue = this.inputValue.slice(0, -1); }
    return { kind: "consumed" };
  }
  if (data.length === 1 && data.charCodeAt(0) >= 32) {
    this.inputValue += data;
    return { kind: "consumed" };
  }
  return { kind: "passthrough" };
}
```

```typescript
// src/wizard/steps/screens.ts
// ScreensStep: stateful add-one-by-one list component for step 3 (Screens).
// D-92: Enter adds screen, Backspace-on-empty removes last, Tab finishes.
// D-94: Minimum 1 screen required to advance (Tab with empty list shows error).
// Returns StepAction to FormPane — Pitfall 5 mitigation.

import type { StepAction } from "./index.ts";
import { truncateToWidth } from "../../canvas/tui-utils.ts";

interface MinimalTheme { fg: (token: string, str: string) => string; }

export class ScreensStep {
  private items: string[] = [];
  private inputValue = "";
  private error: string | null = null;

  constructor(private readonly theme: MinimalTheme) {}

  /** D-98: Pre-populate from spec.screens on re-entry */
  loadFrom(screenNames: string[]): void {
    this.items = [...screenNames];
    this.inputValue = "";
    this.error = null;
  }

  getItems(): string[] { return [...this.items]; }

  handleInput(data: string): StepAction {
    this.error = null;
    if (data === "\r" || data === "\n") {
      const trimmed = this.inputValue.trim();
      if (trimmed.length > 0) { this.items.push(trimmed); this.inputValue = ""; }
      return { kind: "consumed" };
    }
    if (data === "\t") {
      if (this.items.length === 0) {
        this.error = "At least one screen is required.";  // D-94 exact copy
        return { kind: "consumed" };
      }
      return { kind: "advance", args: this.items };
    }
    if (data === "\x7f" || data === "\b") {
      if (this.inputValue.length === 0 && this.items.length > 0) {
        this.items.pop();
      } else {
        this.inputValue = this.inputValue.slice(0, -1);
      }
      return { kind: "consumed" };
    }
    if (data.length === 1 && data.charCodeAt(0) >= 32) {
      this.inputValue += data;
      return { kind: "consumed" };
    }
    return { kind: "passthrough" };
  }

  render(width: number): string[] {
    const lines: string[] = [];
    const placeholder = this.inputValue.length === 0 ? "Screen name..." : "";
    const inputDisplay = this.inputValue || this.theme.fg("muted", placeholder);
    lines.push(truncateToWidth(`> ${inputDisplay}`, width));
    for (const item of this.items) {
      lines.push(truncateToWidth(`  ${item}`, width));
    }
    if (this.error) {
      lines.push(truncateToWidth(this.theme.fg("error", this.error), width));
    }
    return lines;
  }

  invalidate(): void {}
}
```

- Tab: call tryAdvance() (async void)
