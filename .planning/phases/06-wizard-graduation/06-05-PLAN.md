```typescript
export class SpecPreviewPane implements Component {
  private snapshot: Snapshot | null = null;
  private lineCache: string[] | null = null;

  update(snapshot: Snapshot): void {
    this.snapshot = snapshot;
    this.lineCache = null;  // Pitfall 4 prevention
  }

  render(width: number): string[] {
    if (!this.snapshot) return [truncateToWidth("(loading...)", width)];
    if (!this.lineCache) {
      const yamlText = YAML.stringify(this.snapshot.spec);
      this.lineCache = yamlText.split("\n").map(
        (line) => truncateToWidth(line, width - 2),
      );
    }
    return this.lineCache;
  }

  handleInput(_data: string): void {}
  invalidate(): void { this.lineCache = null; }
}
```

```typescript
/**
 * Compute [formWidth, previewWidth] for the wizard 2-pane layout.
 *
 * D-88: 50/50 split (right pane absorbs Math.floor rounding remainder).
 * D-89: Preview collapses to 0 columns when total < 80.

 * @param total - Total terminal width in columns
 * @returns [formWidth, previewWidth]

  if (total < 80) return [total, 0]; // D-89: collapse preview pane
```
